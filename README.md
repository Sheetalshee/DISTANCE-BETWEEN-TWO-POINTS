### PROGRAM:
~~~
#Program to find the distance between two points.
#Developed by:R.Sheetal
#RegisterNumber:23001267
import math
x1=4
y1=2
x2=10
y2=6
a=(x1-x2)*(x1-x2)
b=(y1-y2)*(y1-y2)
c=a+b
sq=math.sqrt(c)
print("{:.2f}".format(sq))
